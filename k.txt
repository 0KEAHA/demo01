document.querySelector('section').addEventListener('mousemove', (e) => {
document.querySelectorAll('.animate-away').forEach(element => { const rect =
element.getBoundingClientRect(); const x = e.clientX - rect.left; const y =
e.clientY - rect.top; const xc = rect.width / 2; const yc = rect.height / 2;
const dx = (xc - 3*x); // Multiply x by 3 to increase distance const dy = (yc -
3*y); // Multiply y by 3 to increase distance element.style.transform =
`translate3d(${dx}px, ${dy}px, 0)`; }); });
document.querySelector('section').addEventListener('mouseleave', () => {
document.querySelectorAll('.animate-away').forEach(element => {
element.style.transform = ''; }); });
